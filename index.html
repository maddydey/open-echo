// ЗВУКОВОЙ ДВИЖОК В СТИЛЕ NINTENDO function playNintendoClick() { if (audioCtx.state === 'suspended') audioCtx.resume(); const now = audioCtx.currentTime; // ПЕРВЫЙ КЛИК (Быстрый щелчок) const osc1 = audioCtx.createOscillator(); const gain1 = audioCtx.createGain(); osc1.type = 'triangle'; // Более мягкий, "игровой" звук osc1.frequency.setValueAtTime(1200, now); gain1.gain.setValueAtTime(0.05, now); gain1.gain.exponentialRampToValueAtTime(0.01, now + 0.05); osc1.connect(gain1); gain1.connect(audioCtx.destination); osc1.start(now); osc1.stop(now + 0.05); // ВТОРОЙ КЛИК (Основной тон с задержкой 0.04с) const osc2 = audioCtx.createOscillator(); const gain2 = audioCtx.createGain(); osc2.type = 'sine'; osc2.frequency.setValueAtTime(800, now + 0.04); gain2.gain.setValueAtTime(0.07, now + 0.04); gain2.gain.exponentialRampToValueAtTime(0.01, now + 0.15); osc2.connect(gain2); gain2.connect(audioCtx.destination); osc2.start(now + 0.04); osc2.stop(now + 0.15); } // ЗВУК ОТПРАВКИ (Похож на звук выхода из приложения в Switch) function playSendSound() { const now = audioCtx.currentTime; const osc = audioCtx.createOscillator(); const gain = audioCtx.createGain(); osc.type = 'sine'; osc.frequency.setValueAtTime(400, now); osc.frequency.exponentialRampToValueAtTime(1200, now + 0.2); // Восходящий тон gain.gain.setValueAtTime(0.05, now); gain.gain.exponentialRampToValueAtTime(0.01, now + 0.2); osc.connect(gain); gain.connect(audioCtx.destination); osc.start(); osc.stop(now + 0.2); }
